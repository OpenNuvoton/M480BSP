255{
256 MSC_T *msc;
257 struct bulk_cb_wrap *cmd_blk; /* MSC Bulk-only command block */
258 int ret;
259
260 //msc_debug_msg("usbh_umas_read - %d, %d\n", sec_no, sec_cnt);
261
262 msc = find_msc_by_drive(drv_no);
263 if (msc == NULL)

272 cmd_blk->Length = 10;
273 cmd_blk->CDB[0] = READ_10;
274 cmd_blk->CDB[1] = msc->lun << 5;

279 cmd_blk->CDB[7] = (sec_cnt >> 8) & 0xFF;
280 cmd_blk->CDB[8] = sec_cnt & 0xFF;

283 if (ret != 0)

419static int umass_init_device(MSC_T *msc)
420{
421 MSC_T *try_msc = msc;
422 struct bulk_cb_wrap *cmd_blk; /* MSC Bulk-only command block */
423 int retries, lun;
424 int8_t bHasMedia = 0;
425 int ret = USBH_ERR_NOT_FOUND;
426
427 for (lun = 0; lun <= msc->max_lun; lun++)
428 {
429 msc_debug_msg("\n******* Read lun %d ******\n\n", lun);
430
431 try_msc->lun = lun;
432 cmd_blk = &try_msc->cmd_blk;
433
434 for (retries = 0; retries < 3; retries++)
435 {
436 if (msc_inquiry(try_msc) == USBH_ERR_STALL)
437 msc_reset(try_msc);
438
439 if (msc_inquiry(try_msc) == USBH_ERR_STALL)
440 msc_reset(try_msc);
441
442 if (msc_test_unit_ready(try_msc) == 0)
443 goto disk_found;
444
445 if (msc_request_sense(try_msc) == 0)
446 {
447 goto disk_found;
448 }
449
450 delay_us(100000); /* delay 100ms, retry later */
451 }
452
453 continue;
454
455disk_found:
456 /*
457 * Valid disk found in this lun. Go...
458 */
459 for (retries = 0; retries < 3; retries++)
460 {
461 msc_debug_msg("READ CAPACITY ==>\n");
462
463 memset(cmd_blk, 0, sizeof(*cmd_blk));
464
465 cmd_blk->Flags = 0x80;
466 cmd_blk->Length = 10;
467 cmd_blk->CDB[0] = READ_CAPACITY;
468 cmd_blk->CDB[1] = lun << 5;
469
470 ret = run_scsi_command(try_msc, try_msc->scsi_buff, 8, 1, 100);
471 if (ret < 0)
472 {
473 msc_debug_msg("READ_CAPACITY failed!\n");
474 if (ret == USBH_ERR_STALL)
475 msc_reset(try_msc);
476 continue;
477 }
478 else
479 break;
480 }
481
482 if (retries >= 3)
483 continue; /* try next lun */
484
485 try_msc->uTotalSectorN = (try_msc->scsi_buff[0] << 24) | (try_msc->scsi_buff[1] << 16) |
486 (try_msc->scsi_buff[2] << 8) | try_msc->scsi_buff[3];
487 try_msc->nSectorSize = (try_msc->scsi_buff[4] << 24) | (try_msc->scsi_buff[5] << 16) |
488 (try_msc->scsi_buff[6] << 8) | try_msc->scsi_buff[7];
489
490 try_msc->drv_no = fatfs_drive_alloc();
491 if (try_msc->drv_no < 0) /* should be failed, unless drive free slot is empty */
492 {
493 ret = USBH_ERR_MEMORY_OUT;
494 break;
495 }
496
497 msc_debug_msg("USB disk [%c] found: size=%d MB, uTotalSectorN=%d\n", msc->drv_no+'0', try_msc->uTotalSectorN / 2048, try_msc->uTotalSectorN);
498
499 msc_list_add(try_msc);
500
501 _path[0] = try_msc->drv_no + '0';
502 f_mount(&try_msc->fatfs_vol, _path, 1);
503 bHasMedia = 1;
504
505
506 /*
507 * duplicate another MSC for next try
508 */
509 try_msc = usbh_alloc_mem(sizeof(*try_msc));
510 if (try_msc == NULL)
511 {
512 ret = USBH_ERR_MEMORY_OUT;
513 break;
514 }
515 memcpy(try_msc, msc, sizeof(*msc));
516 }
517
518 if (bHasMedia)
519 {
520 if (try_msc)
521 usbh_free_mem(try_msc, sizeof(*try_msc));
522 return 0;
523 }
524 return ret;
525}

527static int msc_probe(IFACE_T *iface)
528{
529 ALT_IFACE_T *aif = iface->aif;
530 DESC_IF_T *ifd;
531 MSC_T *msc;
532 int i;
533
534 ifd = aif->ifd;
535
536 /* Is this interface mass storage class? */
537 if (ifd->bInterfaceClass != USB_CLASS_MASS_STORAGE)
538 return USBH_ERR_NOT_MATCHED;
539
540 /* Is supported sub-class? */
541 if ((ifd->bInterfaceSubClass != MSC_SCLASS_SCSI) && (ifd->bInterfaceSubClass != MSC_SCLASS_8070) &&
542 (ifd->bInterfaceSubClass != MSC_SCLASS_RBC))
543 return USBH_ERR_NOT_SUPPORTED;
544
545 /* Is bulk-only protocol? */
546 if (ifd->bInterfaceProtocol != MSC_SPROTO_BULK)
547 {
548 msc_debug_msg("Not bulk-only MSC device!\n");
549 return USBH_ERR_NOT_SUPPORTED;
550 }
551
552 msc = usbh_alloc_mem(sizeof(*msc));
553 if (msc == NULL)
554 return USBH_ERR_MEMORY_OUT;
555 msc->uid = get_ticks();
556
557 /* Find the bulk in and out endpoints */
558 for (i = 0; i < aif->ifd->bNumEndpoints; i++)
559 {
560 if ((aif->ep[i].bmAttributes & EP_ATTR_TT_MASK) == EP_ATTR_TT_BULK)
561 {
562 if ((aif->ep[i].bEndpointAddress & EP_ADDR_DIR_MASK) == EP_ADDR_DIR_IN)
563 msc->ep_bulk_in = &aif->ep[i];
564 else
565 msc->ep_bulk_out = &aif->ep[i];
566 }
567 }
568
569 if ((msc->ep_bulk_in == NULL) || (msc->ep_bulk_out == NULL))
570 {
571 usbh_free_mem(msc, sizeof(*msc));
572 return USBH_ERR_NOT_EXPECTED;
573 }
574
575 msc->iface = iface;
576
577 msc_debug_msg("USB Mass Storage device found. Iface:%d, Alt Iface:%d, bep_in:0x%x, bep_out:0x%x\n", ifd->bInterfaceNumber, ifd->bAlternateSetting, msc->ep_bulk_in->bEndpointAddress, msc->ep_bulk_out->bEndpointAddress);
578
579 get_max_lun(msc);
580
581 return umass_init_device(msc);
582}

584static void msc_disconnect(IFACE_T *iface)
585{
586 int i;
587 MSC_T *msc_p, *msc;
588
589 /*
590 * Remove any hardware EP/QH from Host Controller hardware list.
591 * This will finally result in all transfers aborted.
592 */
593 for (i = 0; i < iface->aif->ifd->bNumEndpoints; i++)
594 {
595 iface->udev->hc_driver->quit_xfer(NULL, &(iface->aif->ep[i]));
596 }
597
598 /*
599 * unmount drive and remove it from MSC device list
600 */
601 msc = g_msc_list;
602 while (msc != NULL)
603 {
604 msc_p = msc->next;
605 if (msc->iface == iface)
606 {
607 fatfs_drive_free(msc->drv_no);
608 msc_list_remove(msc);
609 usbh_free_mem(msc, sizeof(*msc));
610 }
611 msc = msc_p;
612 }
613}
