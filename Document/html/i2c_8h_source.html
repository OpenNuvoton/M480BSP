445{
446 uint32_t u32TimeOutCount = SystemCoreClock; // 1 second timeout
447 (i2c)->CTL0 |= (I2C_CTL0_SI_Msk | I2C_CTL0_STO_Msk);
448 while(i2c->CTL0 & I2C_CTL0_STO_Msk)

450 u32TimeOutCount--;
