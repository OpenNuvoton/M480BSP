287__STATIC_INLINE void RTC_WaitAccessEnable(void)
288{
289 uint32_t u32TimeOutCount = SystemCoreClock; // 1 second timeout
290 uint32_t i = 0;
291
292 while((RTC->RWEN & RTC_RWEN_RTCBUSY_Msk) == RTC_RWEN_RTCBUSY_Msk)
293 {
294 i++;
295 if(i > u32TimeOutCount) break;
296 }
297
298 if(!(SYS->CSERVER & 0x1))
299 {
300 /* To wait RWENF bit is cleared and enable RWENF bit (Access Enable bit) again */
301 RTC->RWEN = RTC_WRITE_KEY;
302 }
303
304 /* To wait RWENF bit is set and user can access the protected-register of RTC from now on */
305 while((RTC->RWEN & RTC_RWEN_RWENF_Msk) == (uint32_t)0x0)
306 {
307 i++;
308 if(i > u32TimeOutCount) break;
309 }
310}
