450static void SendChar_ToUART(int ch)
451{
452 static uint8_t u8Buf[BUF_SIZE] = {0};
453 static int32_t i32Head = 0;
454 static int32_t i32Tail = 0;
455 int32_t i32Tmp;
456
457 /* Only flush the data in buffer to UART when ch == 0 */
458 if(ch)
459 {
460 /* Push char */
461 if(ch == '\n')
462 {
463 i32Tmp = i32Head+1;
464 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
465 if(i32Tmp != i32Tail)
466 {
467 u8Buf[i32Head] = '\r';
468 i32Head = i32Tmp;
469 }
470 }
471
472 i32Tmp = i32Head+1;
473 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
474 if(i32Tmp != i32Tail)
475 {
476 u8Buf[i32Head] = ch;
477 i32Head = i32Tmp;
478 }
479 }
480 else
481 {
482 if(i32Tail == i32Head)
483 return;
484 }
485
486 /* pop char */
487 do
488 {
489 i32Tmp = i32Tail + 1;
490 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
491
492 if((DEBUG_PORT->FIFOSTS & UART_FIFOSTS_TXFULL_Msk) == 0)
493 {
494 DEBUG_PORT->DAT = u8Buf[i32Tail];
495 i32Tail = i32Tmp;
496 }
497 else
498 break; /* FIFO full */
499 }while(i32Tail != i32Head);
500}
