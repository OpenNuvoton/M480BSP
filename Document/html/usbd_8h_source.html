577 for(i = 0ul; i < USBD_MAX_EP; i++)
578 {
579 u32CfgAddr = (uint32_t)(i << 4) + (uint32_t)&USBD->EP[0].CFG; /* USBD_CFG0 */
580 u32Cfg = *((__IO uint32_t *)(u32CfgAddr));
