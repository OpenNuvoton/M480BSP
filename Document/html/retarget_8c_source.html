448static void SendChar_ToUART(int ch)
449{
450 static uint8_t u8Buf[BUF_SIZE] = {0};
451 static int32_t i32Head = 0;
452 static int32_t i32Tail = 0;
453 int32_t i32Tmp;
454
455 /* Only flush the data in buffer to UART when ch == 0 */
456 if(ch)
457 {
458 /* Push char */
459 if(ch == '\n')
460 {
461 i32Tmp = i32Head+1;
462 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
463 if(i32Tmp != i32Tail)
464 {
465 u8Buf[i32Head] = '\r';
466 i32Head = i32Tmp;
467 }
468 }
469
470 i32Tmp = i32Head+1;
471 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
472 if(i32Tmp != i32Tail)
473 {
474 u8Buf[i32Head] = ch;
475 i32Head = i32Tmp;
476 }
477 }
478 else
479 {
480 if(i32Tail == i32Head)
481 return;
482 }
483
484 /* pop char */
485 do
486 {
487 i32Tmp = i32Tail + 1;
488 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
489
490 if((DEBUG_PORT->FIFOSTS & UART_FIFOSTS_TXFULL_Msk) == 0)
491 {
492 DEBUG_PORT->DAT = u8Buf[i32Tail];
493 i32Tail = i32Tmp;
494 }
495 else
496 break; /* FIFO full */
497 }while(i32Tail != i32Head);
