450/* Non-block implement of send char */
451#define BUF_SIZE 2048
452static void SendChar_ToUART(int ch)
453{
454 static uint8_t u8Buf[BUF_SIZE] = {0};
455 static int32_t i32Head = 0;
456 static int32_t i32Tail = 0;
457 int32_t i32Tmp;
458
459 /* Only flush the data in buffer to UART when ch == 0 */
460 if(ch)
461 {
462 /* Push char */
463 if(ch == '\n')
464 {
465 i32Tmp = i32Head+1;
466 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
467 if(i32Tmp != i32Tail)
468 {
469 u8Buf[i32Head] = '\r';
470 i32Head = i32Tmp;
471 }
472 }
473
474 i32Tmp = i32Head+1;
475 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
476 if(i32Tmp != i32Tail)
477 {
478 u8Buf[i32Head] = ch;
479 i32Head = i32Tmp;
480 }
481 }
482 else
483 {
484 if(i32Tail == i32Head)
485 return;
486 }
487
488 /* pop char */
489 do
490 {
491 i32Tmp = i32Tail + 1;
492 if(i32Tmp > BUF_SIZE) i32Tmp = 0;
493
494 if((DEBUG_PORT->FIFOSTS & UART_FIFOSTS_TXFULL_Msk) == 0)
495 {
496 DEBUG_PORT->DAT = u8Buf[i32Tail];
497 i32Tail = i32Tmp;
498 }
499 else
500 break; /* FIFO full */
501 }while(i32Tail != i32Head);
502}
